// Cambia entre los formularios de login y registro
function toggleForm() {
  const loginForm = document.getElementById("login-form");
  const registerForm = document.getElementById("register-form");
  const formText = document.getElementById("form-text");

  const isLoginVisible = loginForm.style.display !== "none";

  // Alterna entre mostrar y ocultar cada formulario
  loginForm.style.display = isLoginVisible ? "none" : "block";
  registerForm.style.display = isLoginVisible ? "block" : "none";

  // Cambia el texto del t√≠tulo seg√∫n el formulario 
  formText.innerText = isLoginVisible ? "Registrarse" : "Iniciar Sesi√≥n";

  // Limpia cualquier mensaje anterior
  document.getElementById("message").innerText = "";
}

// Muestra/oculta la contrase√±a en los campos de passwor
function togglePassword(id, icon) {
  const input = document.getElementById(id);
  if (input.type === "password") {
    input.type = "text"; // Mostrar contrase√±a
    icon.classList.replace("fa-eye", "fa-eye-slash");
  } else {
    input.type = "password"; // Ocultar contrase√±a
    icon.classList.replace("fa-eye-slash", "fa-eye");
  }
}

// Valida si un email tiene el formato correcto
function isValidEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

// Registra un nuevo usuario y lo guarda en localStorage
function register() {
  const username = document.getElementById("register-username").value;
  const email = document.getElementById("register-email").value;
  const password = document.getElementById("register-password").value;
  const terms = document.getElementById("terms").checked;
  const msg = document.getElementById("message");

  // Verifica que los campos no est√©n vac√≠os y que se hayan aceptado los t√©rminos
  if (!username || !email || !password || !terms) {
    msg.innerText = "Completa todos los campos.";
    return;
  }

  // Verifica si el email es v√°lido
  if (!isValidEmail(email)) {
    msg.innerText = "Email no v√°lido.";
    return;
  }

  // Crea un objeto con los datos del usuario
  const user = { username, email, password };

  // guarda los datos del usuario en localStorage
  localStorage.setItem("userData", JSON.stringify(user));

  // muestra mensaje y cambia al formulario del login
  msg.innerText = "Guardar contrase√±a. Ya est√°s registrado.";
  toggleForm();
}

// verifica si los datos ingresados coinciden con los del usuario guardado
function login() {
  const emailInput = document.getElementById("login-email").value;
  const password = document.getElementById("login-password").value;
  const msg = document.getElementById("message");

  // obtiene los datos del usuario guardado en el localStorage
  const saved = JSON.parse(localStorage.getItem("userData"));

  // Compara email/usuario y contrase√±a
  if (saved && (emailInput === saved.email || emailInput === saved.username) && password === saved.password) {
    msg.innerText = "Sesi√≥n iniciada correctamente.";
    // Podr√≠as redirigir a otra p√°gina con:
    // window.location.href = "inicio.html";
  } else {
    msg.innerText = "Contrase√±a o email incorrecto.";
  }
}
